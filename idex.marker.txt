<!-- เอา onclick ออก แล้วใช้ id="start-btn" แทน เพื่อไม่ให้ GitHub Pages บล็อกความปลอดภัย -->
<button type="button" id="start-btn">Start</button>
<div id="status-text" class="status-text"></div>
<div id="webcam-container"></div>
<div id="label-container"></div>

<script src="https://cdn.jsdelivr.net/npm/@tensorflow/tfjs@latest/dist/tf.min.js"></script>
<script src="https://cdn.jsdelivr.net/npm/@teachablemachine/image@latest/dist/teachablemachine-image.min.js"></script>
<script type="text/javascript">
    // ลิงก์โมเดลที่มีคลาส mask, nomask, openmask
    const URL = "https://teachablemachine.withgoogle.com/models/6zKJgPS1_/";

    let model, webcam, labelContainer, maxPredictions;
    let predictInterval;

    // ใช้ addEventListener ดักจับการกดปุ่มแทน เพื่อให้รันบน GitHub Pages ได้ฉลุย
    document.getElementById("start-btn").addEventListener("click", init);

    async function init() {
        const startBtn = document.getElementById("start-btn");
        const statusText = document.getElementById("status-text");
        
        startBtn.disabled = true;
        statusText.innerText = "Loading model...";

        try {
            const modelURL = URL + "model.json";
            const metadataURL = URL + "metadata.json";

            model = await tmImage.load(modelURL, metadataURL);
            maxPredictions = model.getTotalClasses();

            statusText.innerText = "Starting webcam...";

            const flip = true; 
            webcam = new tmImage.Webcam(200, 200, flip); 
            await webcam.setup(); 
            await webcam.play();
            
            statusText.innerText = "Running (Predicting every 3 seconds)";

            document.getElementById("webcam-container").appendChild(webcam.canvas);
            labelContainer = document.getElementById("label-container");
            
            // เคลียร์ค่าเก่าและสร้าง div รอแสดงผลลัพธ์
            labelContainer.innerHTML = "";
            for (let i = 0; i < maxPredictions; i++) {
                labelContainer.appendChild(document.createElement("div"));
            }

            // สั่งทำนายรอบแรกทันที
            updateAndPredict();

            // ตั้งเวลาให้ดึงภาพจากกล้องไปทำนายผลใหม่ "ทุกๆ 3 วินาที" ตามต้องการ
            predictInterval = setInterval(updateAndPredict, 3000);

        } catch (error) {
            console.error(error);
            statusText.innerText = "Error: Cannot access camera or model.";
            startBtn.disabled = false;
        }
    }

    // ฟังก์ชันอัปเดตภาพและส่งไปทายผล
    async function updateAndPredict() {
        if (webcam) {
            webcam.update(); 
            await predict(); 
        }
    }

    // ฟังก์ชันทำนายผลจริงจากโมเดล
    async function predict() {
        const prediction = await model.predict(webcam.canvas);
        for (let i = 0; i < maxPredictions; i++) {
            const classPrediction =
                prediction[i].className + ": " + prediction[i].probability.toFixed(2);
            labelContainer.childNodes[i].innerHTML = classPrediction;
        }
    }
</script>
